2. 编码技巧：一个位姿下有多个测点，每个测点都有计算，所有测点计算完后，通知出去，写法技巧：用lamda的defer，可以很方便解决这个问题，该方法可以推广到其他场景

```cpp
class Camera {
public:
    virtual void move()=0;
    virtual void zoom()=0;
    virtual void setStrategy(CameraStrategy* strategy)=0;
};

class CameraStrategy {
public:
    virtual void move()=0;
    virtual void zoom()=0;
};

class Camera2DStrategy : public CameraStrategy {
public:
    void move() override { /* move 2D camera */ }
    void zoom() override { /* zoom 2D camera */ }
    void rotate() { /* rotate 2D camera */ }
};

class Camera3DStrategy : public CameraStrategy {
public:
    void move() override { /* move 3D camera */ }
    void zoom() override { /* zoom 3D camera */ }
    void fly() { /* fly 3D camera */ }
};

class AbstractCamera : public Camera {
private:
    CameraStrategy* strategy2D;
    CameraStrategy* strategy3D;
public:
    void setStrategy(CameraStrategy* strategy) override { 
        if(dynamic_cast<Camera2DStrategy*>(strategy))
            strategy2D = strategy;
        else if(dynamic_cast<Camera3DStrategy*>(strategy))
            strategy3D = strategy;
    }
    void move() override { 
        if(strategy2D) strategy2D->move();
        else if(strategy3D) strategy3D->move();
    }
    void zoom() override { 
        if(strategy2D) strategy2D->zoom();
        else if(strategy3D) strategy3D->zoom();
    }
    void rotate() {
        if(strategy2D) dynamic_cast<Camera2DStrategy*>(strategy2D)->rotate();
    }
    void fly() {
        if(strategy3D) dynamic_cast<Camera3DStrategy*>(strategy3D)->fly();
    }
};


```
