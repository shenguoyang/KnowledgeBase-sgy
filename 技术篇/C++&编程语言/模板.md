class FieldObject;
typedef std::shared_ptr<FieldObject> FieldObjectSPtr;
class FieldObject
{
public:
	typedef memory::Vector<memory::String>			VectorNames;
	typedef memory::Vector<database::DBFieldType>	VectorTypes;
	typedef memory::Vector<database::DBValueWrite>	VectorValues;
public:
	FieldObject(std::string prefix = "")
		: m_prefix(std::move(prefix))
	{
		
	}
	virtual ~FieldObject()
	{
	}

public:
	template<typename T>
	static int32_t parseQueryResultList(FieldObjectSPtr obj, const isv::database::DBQueryResultPtr& result, T& v)
	{
		while (isv::database::DATABASE_ERROR_NO == result->fetchNext())
		{
			typename T::value_type temp;

			auto ret = obj->parseQueryResult(result, &temp);
			if (ret != isv::database::DATABASE_ERROR_NO)
			{
				return ret;
			}

			v.push_back(std::move(temp));
		}

		return isv::database::DATABASE_ERROR_NO;
	}

	template<typename T>
	static int32_t parseSingleQueryResult(FieldObjectSPtr obj, const isv::database::DBQueryResultPtr& result, T& v)
	{
		auto ret = result->fetchNext();
		if (isv::database::DATABASE_ERROR_NO == ret)
		{
			ret = obj->parseQueryResult(result, &v);
		}

		return ret;
	}

public:
	virtual	void getTableHead(VectorNames& names, VectorTypes& types) = 0;
	virtual	void generateTableData(database::DBTable& tblData, uint32_t rowIndex, const void* v) = 0
	virtual	void generateUpdateFields(VectorNames& names, VectorValues& values, const void* v) = 0;
	virtual int32_t parseQueryResult(const isv::database::DBQueryResultPtr& result, void* v) = 0;
public:
	void setPrefix(std::string prefix) { m_prefix = std::move(prefix); }
protected:
	FieldObject(const FieldObject&) = delete;
	FieldObject& operator=(const FieldObject&) = delete;
protected:
	std::string	m_prefix;
};

}}
