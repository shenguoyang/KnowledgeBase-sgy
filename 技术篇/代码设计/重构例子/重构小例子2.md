### <font style="color:rgb(64, 64, 64);">初始代码：混乱的条件和重复参数</font>
<font style="color:rgb(64, 64, 64);">假设有一个订单处理系统，原始代码如下：</font>

```cpp
#include <string>
#include <vector>
#include <ctime>

class Order {
public:
// 构造函数包含大量重复参数
Order(const std::string& customerName, 
const std::string& customerAddress,
const std::string& shippingMethod,
double productWeight,
bool isExpress,
time_t orderDate) 
: customerName_(customerName),
customerAddress_(customerAddress),
shippingMethod_(shippingMethod),
productWeight_(productWeight),
isExpress_(isExpress),
orderDate_(orderDate) {}

// 计算运费和税费的逻辑复杂
double calculateTotalCost() {
    double shippingCost = 0;

    // 运费计算（包含大量条件）
    if (shippingMethod_ == "Standard") {
        shippingCost = productWeight_ * 0.5;
        if (isExpress_) shippingCost += 10;
    } else if (shippingMethod_ == "Premium") {
        shippingCost = productWeight_ * 0.8 + 15;
    } else {
        shippingCost = 20; // 默认费用
    }

    // 税费计算（条件与运费耦合）
    double tax = 0;
    if (customerAddress_.find("NY") != std::string::npos) {
        tax = shippingCost * 0.08;
    } else if (customerAddress_.find("CA") != std::string::npos) {
        tax = shippingCost * 0.09;
    }

    return shippingCost + tax;
}

private:
std::string customerName_;
std::string customerAddress_;
std::string shippingMethod_;
double productWeight_;
bool isExpress_;
time_t orderDate_;
};
```

### <font style="color:rgb(64, 64, 64);">重构步骤</font>
#### <font style="color:rgb(64, 64, 64);">第一步：引入参数对象（Introduce Parameter Object）</font>
**<font style="color:rgb(64, 64, 64);">问题</font>**<font style="color:rgb(64, 64, 64);">：构造函数参数过多，且</font><font style="color:rgb(64, 64, 64);"> </font>`<font style="color:rgb(64, 64, 64);">calculateTotalCost</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">中重复使用地址和物流方式等参数。</font>

**<font style="color:rgb(64, 64, 64);">解决</font>**<font style="color:rgb(64, 64, 64);">：将相关参数封装为 </font>`<font style="color:rgb(64, 64, 64);">CustomerInfo</font>`<font style="color:rgb(64, 64, 64);"> 和 </font>`<font style="color:rgb(64, 64, 64);">ShippingInfo</font>`<font style="color:rgb(64, 64, 64);"> 对象。</font>

```cpp
// 新增参数对象
struct CustomerInfo {
std::string name;
std::string address;
};

struct ShippingInfo {
std::string method;
double productWeight;
bool isExpress;
time_t orderDate;
};

// 修改 Order 类
class Order {
public:
Order(const CustomerInfo& customer, const ShippingInfo& shipping)
: customer_(customer), shipping_(shipping) {}

double calculateTotalCost() { /* 后续优化 */ }

private:
CustomerInfo customer_;
ShippingInfo shipping_;
};
```

#### <font style="color:rgb(64, 64, 64);">第二步：分解条件表达式（Decompose Conditional）</font>
**<font style="color:rgb(64, 64, 64);">问题</font>**<font style="color:rgb(64, 64, 64);">：</font>`<font style="color:rgb(64, 64, 64);">calculateTotalCost</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">中的条件逻辑复杂，难以维护。</font>

**<font style="color:rgb(64, 64, 64);">解决</font>**<font style="color:rgb(64, 64, 64);">：将运费和税费计算拆分为独立方法，并提取策略类。</font>

```cpp
// 在 Order 类中添加方法：
double calculateShippingCost() {
    if (shipping_.method == "Standard") {
        return calculateStandardShipping();
    } else if (shipping_.method == "Premium") {
        return calculatePremiumShipping();
    } else {
        return 20.0; // 默认
    }
}

double calculateStandardShipping() {
    double cost = shipping_.productWeight * 0.5;
    if (shipping_.isExpress) cost += 10;
    return cost;
}

double calculatePremiumShipping() {
    return shipping_.productWeight * 0.8 + 15;
}

// 税费计算单独提取
double calculateTax(double shippingCost) {
    if (customer_.address.find("NY") != std::string::npos) {
        return shippingCost * 0.08;
    } else if (customer_.address.find("CA") != std::string::npos) {
        return shippingCost * 0.09;
    }
    return 0;
}

// 最终方法
double calculateTotalCost() {
    double shippingCost = calculateShippingCost();
    double tax = calculateTax(shippingCost);
    return shippingCost + tax;
}
```

#### <font style="color:rgb(64, 64, 64);">第三步：策略模式替换条件逻辑（Replace Conditional with Strategy）</font>
**<font style="color:rgb(64, 64, 64);">问题</font>**<font style="color:rgb(64, 64, 64);">：运费计算仍包含硬编码条件，扩展新物流方式需修改</font><font style="color:rgb(64, 64, 64);"> </font>`<font style="color:rgb(64, 64, 64);">Order</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">类。</font>

**<font style="color:rgb(64, 64, 64);">解决</font>**<font style="color:rgb(64, 64, 64);">：定义 </font>`<font style="color:rgb(64, 64, 64);">ShippingStrategy</font>`<font style="color:rgb(64, 64, 64);"> 抽象接口，将不同物流策略分离。</font>

```cpp
// 定义运费策略接口
class ShippingStrategy {
public:
virtual ~ShippingStrategy() = default;
virtual double calculate(const ShippingInfo& info) const = 0;
};

// 具体策略类
class StandardShipping : public ShippingStrategy {
public:
double calculate(const ShippingInfo& info) const override {
    double cost = info.productWeight * 0.5;
    if (info.isExpress) cost += 10;
    return cost;
}
};

class PremiumShipping : public ShippingStrategy {
public:
double calculate(const ShippingInfo& info) const override {
    return info.productWeight * 0.8 + 15;
}
};

// 修改 Order 类
class Order {
public:
Order(const CustomerInfo& customer, 
const ShippingInfo& shipping,
std::unique_ptr<ShippingStrategy> strategy)
: customer_(customer), 
shipping_(shipping),
strategy_(std::move(strategy)) {}

double calculateShippingCost() {
    return strategy_->calculate(shipping_);
}

private:
std::unique_ptr<ShippingStrategy> strategy_;
// ... 其他成员
};
```

### <font style="color:rgb(64, 64, 64);">最终重构代码</font>
```cpp
#include <string>
#include <memory>
#include <ctime>

// 参数对象
struct CustomerInfo { std::string name, address; };
struct ShippingInfo { std::string method; double productWeight; bool isExpress; time_t orderDate; };

// 运费策略抽象
class ShippingStrategy {
public:
virtual ~ShippingStrategy() = default;
virtual double calculate(const ShippingInfo& info) const = 0;
};

// 具体策略
class StandardShipping : public ShippingStrategy { /* 同上 */ };
class PremiumShipping : public ShippingStrategy { /* 同上 */ };

// 税费策略（可选扩展）
class TaxStrategy {
public:
virtual double calculate(const CustomerInfo& customer, double shippingCost) const = 0;
};

class NYTaxStrategy : public TaxStrategy {
public:
double calculate(const CustomerInfo& customer, double shippingCost) const override {
    return shippingCost * 0.08;
}
};

// Order 类（最终版）
class Order {
public:
Order(const CustomerInfo& customer,
const ShippingInfo& shipping,
std::unique_ptr<ShippingStrategy> shippingStrategy,
std::unique_ptr<TaxStrategy> taxStrategy)
: customer_(customer),
shipping_(shipping),
shippingStrategy_(std::move(shippingStrategy)),
taxStrategy_(std::move(taxStrategy)) {}

double calculateTotalCost() {
    double shippingCost = shippingStrategy_->calculate(shipping_);
    double tax = taxStrategy_->calculate(customer_, shippingCost);
    return shippingCost + tax;
}

private:
CustomerInfo customer_;
ShippingInfo shipping_;
std::unique_ptr<ShippingStrategy> shippingStrategy_;
std::unique_ptr<TaxStrategy> taxStrategy_;
};

// 使用示例
int main() {
    CustomerInfo customer{"Alice", "123 Main St, NY"};
    ShippingInfo shipping{"Standard", 5.0, true, time(nullptr)};

    auto order = Order(
    customer,
    shipping,
    std::make_unique<StandardShipping>(),
    std::make_unique<NYTaxStrategy>()
    );

    double total = order.calculateTotalCost(); // 计算结果
    return 0;
}
```

### <font style="color:rgb(64, 64, 64);">总结</font>
+ **<font style="color:rgb(64, 64, 64);">参数对象</font>**<font style="color:rgb(64, 64, 64);">：消除重复参数，提升代码可读性。</font>
+ **<font style="color:rgb(64, 64, 64);">策略模式</font>**<font style="color:rgb(64, 64, 64);">：将易变的运费/税费逻辑分离，符合开闭原则。</font>
+ **<font style="color:rgb(64, 64, 64);">职责分离</font>**<font style="color:rgb(64, 64, 64);">：</font>`<font style="color:rgb(64, 64, 64);">Order</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">类不再负责具体计算，仅协调策略。</font>

<font style="color:rgb(64, 64, 64);">此案例展示了如何通过</font>**<font style="color:rgb(64, 64, 64);">组合对象</font>**<font style="color:rgb(64, 64, 64);">和</font>**<font style="color:rgb(64, 64, 64);">行为抽象</font>**<font style="color:rgb(64, 64, 64);">解决复杂条件问题，与之前的继承方案形成鲜明对比。</font>
