## 本例优点
1. **<font style="color:rgb(64, 64, 64);">明确职责</font>**<font style="color:rgb(64, 64, 64);">：</font>
    - `<font style="color:rgb(64, 64, 64);">Movie</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">负责影片类型的行为。</font>
    - `<font style="color:rgb(64, 64, 64);">Rental</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">负责租赁相关的计算。</font>
    - `<font style="color:rgb(64, 64, 64);">Customer</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">负责生成账单。</font>
2. **<font style="color:rgb(64, 64, 64);">扩展性强</font>**<font style="color:rgb(64, 64, 64);">：</font>
    - <font style="color:rgb(64, 64, 64);">新增影片类型时，只需继承</font><font style="color:rgb(64, 64, 64);"> </font>`<font style="color:rgb(64, 64, 64);">Movie</font>`<font style="color:rgb(64, 64, 64);"> </font><font style="color:rgb(64, 64, 64);">并实现虚函数，无需修改现有代码。</font>
3. **<font style="color:rgb(64, 64, 64);">可读性高</font>**<font style="color:rgb(64, 64, 64);">：</font>
    - <font style="color:rgb(64, 64, 64);">消除了复杂的条件逻辑，代码更易理解和维护。</font>
4. **<font style="color:rgb(64, 64, 64);">符合 SOLID 原则</font>**<font style="color:rgb(64, 64, 64);">：</font>
    - <font style="color:rgb(64, 64, 64);">单一职责、开闭原则、依赖倒置等设计原则得到体现。</font>

### <font style="color:rgb(64, 64, 64);">初始代码：混乱的条件逻辑</font>
```cpp
#include <string>
#include <vector>

class Movie {
public:
enum Type { REGULAR, NEW_RELEASE, CHILDRENS };
Movie(Type type, const std::string& title) : type_(type), title_(title) {}
Type getType() const { return type_; }
std::string getTitle() const { return title_; }

private:
Type type_;
std::string title_;
};

class Rental {
public:
Rental(const Movie& movie, int daysRented) 
: movie_(movie), daysRented_(daysRented) {}
const Movie& getMovie() const { return movie_; }
int getDaysRented() const { return daysRented_; }

private:
Movie movie_;
int daysRented_;
};

class Customer {
public:
Customer(const std::string& name) : name_(name) {}
void addRental(const Rental& rental) { rentals_.push_back(rental); }

// 生成账单的原始方法（包含大量条件判断）
std::string statement() {
    double totalAmount = 0;
    int frequentRenterPoints = 0;
    std::string result = "Rental Record for " + name_ + "\n";

    for (const Rental& rental : rentals_) {
        double thisAmount = 0;
        Movie::Type type = rental.getMovie().getType();

        // 计算租金的核心逻辑（条件分支）
        switch (type) {
            case Movie::REGULAR:
                thisAmount += 2;
                if (rental.getDaysRented() > 2)
                    thisAmount += (rental.getDaysRented() - 2) * 1.5;
                break;
            case Movie::NEW_RELEASE:
                thisAmount += rental.getDaysRented() * 3;
                break;
            case Movie::CHILDRENS:
                thisAmount += 1.5;
                if (rental.getDaysRented() > 3)
                    thisAmount += (rental.getDaysRented() - 3) * 1.5;
                break;
        }

        // 计算积分
        frequentRenterPoints++;
        if (type == Movie::NEW_RELEASE && rental.getDaysRented() > 1)
            frequentRenterPoints++;

        // 添加明细
        result += "\t" + rental.getMovie().getTitle() + "\t" 
            + std::to_string(thisAmount) + "\n";
        totalAmount += thisAmount;
    }

    result += "Amount owed is " + std::to_string(totalAmount) + "\n";
    result += "You earned " + std::to_string(frequentRenterPoints) 
        + " frequent renter points";
    return result;
}

private:
std::string name_;
std::vector<Rental> rentals_;
};
```

#### <font style="color:rgb(64, 64, 64);">第一步：提取方法（Extract Method）</font>
**<font style="color:rgb(64, 64, 64);">目标</font>**<font style="color:rgb(64, 64, 64);">：将 </font>`<font style="color:rgb(64, 64, 64);">statement()</font>`<font style="color:rgb(64, 64, 64);"> 中的金额计算逻辑抽离到独立方法中。</font>

```cpp
// 在 Customer 类中添加方法：
double getAmountForRental(const Rental& rental) {

        case Movie::REGULAR:

            break;

            if (rental.getDaysRented() > 3)

}
```

### <font style="color:rgb(64, 64, 64);">完整重构代码</font>
```cpp
#include <string>
#include <vector>
#include <memory>
#include <iostream>

// 抽象基类 Movie
class Movie {
public:
virtual ~Movie() = default;
virtual double getCharge(int daysRented) const = 0;
virtual int getFrequentRenterPoints(int daysRented) const { return 1; }
std::string getTitle() const { return title_; }
protected:
Movie(const std::string& title) : title_(title) {}
private:
std::string title_;
};

// 具体影片类型
class RegularMovie : public Movie {
public:
RegularMovie(const std::string& title) : Movie(title) {}
double getCharge(int daysRented) const override {
    double result = 2;
    if (daysRented > 2) result += (daysRented - 2) * 1.5;
    return result;
}
};

class NewReleaseMovie : public Movie {
public:
NewReleaseMovie(const std::string& title) : Movie(title) {}
double getCharge(int daysRented) const override {
    return daysRented * 3;
}
int getFrequentRenterPoints(int daysRented) const override {
    return (daysRented > 1) ? 2 : 1;
}
};

class ChildrensMovie : public Movie {
public:
ChildrensMovie(const std::string& title) : Movie(title) {}
double getCharge(int daysRented) const override {
    double result = 1.5;
    if (daysRented > 3) result += (daysRented - 3) * 1.5;
    return result;
}
};

// Rental 类
class Rental {
public:
Rental(std::shared_ptr<Movie> movie, int daysRented)
: movie_(movie), daysRented_(daysRented) {}
std::shared_ptr<Movie> getMovie() const { return movie_; }
int getDaysRented() const { return daysRented_; }
double getCharge() const {
    return movie_->getCharge(daysRented_);
}
int getFrequentRenterPoints() const {
    return movie_->getFrequentRenterPoints(daysRented_);
}

private:
std::shared_ptr<Movie> movie_;
int daysRented_;
};

// Customer 类
class Customer {
public:
Customer(const std::string& name) : name_(name) {}
void addRental(const Rental& rental) { rentals_.push_back(rental); }

std::string statement() {
    double totalAmount = 0;
    int frequentRenterPoints = 0;
    std::string result = "Rental Record for " + name_ + "\n";

    for (const Rental& rental : rentals_) {
        double thisAmount = rental.getCharge();
        frequentRenterPoints += rental.getFrequentRenterPoints();

        // 添加明细
        result += "\t" + rental.getMovie()->getTitle() + "\t"
            + std::to_string(thisAmount) + "\n";
        totalAmount += thisAmount;
    }

    // 添加总计
    result += "Amount owed is " + std::to_string(totalAmount) + "\n";
    result += "You earned " + std::to_string(frequentRenterPoints)
        + " frequent renter points\n";
    return result;
}

private:
std::string name_;
std::vector<Rental> rentals_;
};

// 测试代码
int main() {
    // 创建影片
    std::shared_ptr<Movie> movie1 = std::make_shared<RegularMovie>("The Godfather");
    std::shared_ptr<Movie> movie2 = std::make_shared<NewReleaseMovie>("Avengers: Endgame");
    std::shared_ptr<Movie> movie3 = std::make_shared<ChildrensMovie>("Toy Story");

    // 创建租赁
    Rental rental1(movie1, 3);
    Rental rental2(movie2, 2);
    Rental rental3(movie3, 4);

    // 创建客户并添加租赁
    Customer customer("John Doe");
    customer.addRental(rental1);
    customer.addRental(rental2);
    customer.addRental(rental3);

    // 打印账单
    std::cout << customer.statement() << std::endl;

    return 0;
}
```
