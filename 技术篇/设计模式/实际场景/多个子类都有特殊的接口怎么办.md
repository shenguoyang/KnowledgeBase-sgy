```cpp
// ==================== 核心接口 ====================
class IImageCapture {
public:
    virtual ~IImageCapture() = default;
    virtual bool startCapture() = 0;
    virtual void stopCapture() = 0;
	virtual CaptureMode getMode() const = 0;
};

// 在线检测模式特有接口
class IOnlineDetection {
public:
    virtual ~IOnlineDetection() = default;
    virtual void setDetectionParams(const DetectionParams& params) = 0;
   
};

// 在线检测模式实现
class OnlineDetectionCapture : public IImageCapture, public IOnlineDetection {
    // 同时实现两个接口
    bool startCapture() override { /* 实现 */ }
    void stopCapture() override { /* 实现 */ }
     CaptureMode getMode() const override { return CaptureMode::ONLINE_DETECTION; }
    // 特有接口实现
    void setDetectionParams(const DetectionParams& params) override { /* 实现 */ }
};

// ==================== 控制器封装 ====================
class CaptureController {
private:
    std::unique_ptr<IImageCapture> currentCapture;
    
    // 辅助方法：获取在线检测接口
    IOnlineDetection* getOnlineDetectionInterface() {
        if (!currentCapture || currentCapture->getMode() != CaptureMode::ONLINE_DETECTION) {
            return nullptr;
        }
        return dynamic_cast<IOnlineDetection*>(currentCapture.get());
    }
    
public:
    // 基础操作
    void switchToMode(CaptureMode mode) {
        // 创建对应模式的capture
        currentCapture = createCapture(mode);
    }
    
    bool start() {
        return currentCapture ? currentCapture->startCapture() : false;
    }
    
    void stop() {
        if (currentCapture) currentCapture->stopCapture();
    }
    
    // ========== 封装在线检测特有操作 ==========
    
    // 方式1：类型安全的特定方法
    bool setDetectionParamsForOnlineMode(const DetectionParams& params) {
        if (auto detection = getOnlineDetectionInterface()) {
            detection->setDetectionParams(params);
            return true;
        }
        return false; // 当前不是在线检测模式
    }
};

// ==================== 使用示例 ====================
int main() {
    CaptureController controller;
    
    // 1. 基础使用
    controller.switchToMode(CaptureMode::ONLINE_DETECTION);
    controller.start();
    
    // 2. 设置在线检测特有参数
    DetectionParams params;
    params.confidenceThreshold = 0.8;
    params.detectionInterval = 1000;
    
    // 直接调用
    controller.setDetectionParamsForOnlineMode(params);
    return 0;
}
```

方案3：命令模式

精简代码实现

```cpp
// ==================== 命令接口 ====================
class ICaptureCommand {
public:
    virtual ~ICaptureCommand() = default;
    virtual bool execute(IImageCapture* target) = 0;
    virtual std::string getName() const = 0;
};

// ==================== 具体命令 ====================
// 基础命令
class StartCaptureCommand : public ICaptureCommand {
public:
    bool execute(IImageCapture* target) override {
        return target->startCapture();
    }
    std::string getName() const override { return "StartCapture"; }
};

class StopCaptureCommand : public ICaptureCommand {
public:
    bool execute(IImageCapture* target) override {
        target->stopCapture();
        return true;
    }
    std::string getName() const override { return "StopCapture"; }
};

// 在线检测特有命令
class setDetectionParamsCommand : public ICaptureCommand {
private:
    std::string modelPath;
    
public:
    explicit setDetectionParamsCommand(const std::string& path) 
        : modelPath(path) {}
    
    bool execute(IImageCapture* target) override {
        auto detection = dynamic_cast<OnlineDetectionCapture*>(target);
        if (!detection) {
            std::cerr << "错误：目标不是OnlineDetectionCapture" << std::endl;
            return false;
        }
        detection->setDetectionParams(modelPath);
        return true;
    }
    
    std::string getName() const override {
        return "SetDetectionModel(" + modelPath + ")";
    }
};


// ==================== 命令执行器 ====================
class CaptureCommandInvoker {
private:
    IImageCapture* target;
    std::vector<std::unique_ptr<ICaptureCommand>> executedCommands;
    
public:
    explicit CaptureCommandInvoker(IImageCapture* captureTarget) 
        : target(captureTarget) {}
    
    // 执行单个命令
    bool executeCommand(std::unique_ptr<ICaptureCommand> command) {
        if (!target || !command) return false;
        
        bool success = command->execute(target);
        if (success) {
            executedCommands.push_back(std::move(command));
            std::cout << "命令执行成功: " 
                      << executedCommands.back()->getName() << std::endl;
        }
        return success;
    }
};

// ==================== 使用示例 ====================
int main() {
    // 创建目标对象
    auto capture = std::make_unique<OnlineDetectionCapture>();
    
    // 创建命令执行器
    CaptureCommandInvoker invoker(capture.get());
    
    // 示例1：执行基础命令
    invoker.executeCommand(std::make_unique<StartCaptureCommand>());
    
    // 示例2：执行在线检测特有命令
    invoker.executeCommand(std::make_unique<SetDetectionModelCommand>("model.onnx"));
    invoker.executeCommand(std::make_unique<SetROICommand>(std::vector<Rect>{{0,0,100,100}}));
    
    // 示例3：执行可撤销命令
    invoker.executeCommand(std::make_unique<AdjustThresholdCommand>(0.8));
    invoker.undoLastCommand(); // 撤销阈值调整
    
    // 示例4：批量执行
    invoker.executeCommands(
        std::make_unique<SetDetectionModelCommand>("model_v2.onnx"),
        std::make_unique<StartCaptureCommand>()
    );
    
    // 示例5：使用组合命令
    auto setupCommand = std::make_unique<CompositeCommand>("OnlineDetectionSetup");
    setupCommand->addCommand(std::make_unique<SetDetectionModelCommand>("model.onnx"));
    setupCommand->addCommand(std::make_unique<SetROICommand>(std::vector<Rect>{{0,0,200,200}}));
    setupCommand->addCommand(std::make_unique<AdjustThresholdCommand>(0.75));
    
    invoker.executeCommand(std::move(setupCommand));
    
    // 查看执行历史
    invoker.printCommandHistory();
    
    return 0;
}
```

对比分析与适用场景

方案2：控制器封装

```cpp
// 特点：集中管理，简单直接
controller.switchToMode(ONLINE_DETECTION)
    .withModelPath("model.onnx")
    .withROI(rois)
    .start();
```

优点：

1. 简单直观：API调用像自然语言
2. 类型安全：编译时检查，运行时类型转换在控制器内部处理
3. 使用方便：上层只需与控制器交互
4. 代码集中：所有特殊操作逻辑集中在控制器中

缺点：

1. 控制器膨胀：随着模式增多，控制器代码会变得庞大
2. 不易扩展：新增模式需要修改控制器
3. 职责过重：控制器承担了太多职责

适用场景：

· ✅ 模式数量固定（2-3种）  
· ✅ 特殊操作数量不多  
· ✅ 系统规模不大  
· ✅ 需要简洁的API

```cpp
// 工控设备配置

```

方案3：命令模式

```cpp
// 特点：灵活可扩展，支持复杂操作
invoker.executeCommand(make_unique<SetModelCommand>("model.onnx"));
invoker.executeCommand(make_unique<SetROICommand>(rois));
```

优点：

1. 高度解耦：命令与接收者解耦
2. 易于扩展：新增命令无需修改现有代码
3. 支持复杂操作：组合命令、撤销/重做
4. 支持队列/日志：命令可序列化、排队执行
5. 支持宏操作：可录制和回放操作序列

缺点：

1. 复杂度高：需要定义大量命令类
2. 过度设计：简单场景显得笨重
3. 性能开销：命令对象创建和销毁开销

适用场景：

· ✅ 需要撤销/重做功能  
· ✅ 需要记录操作历史  
· ✅ 操作需要排队执行  
· ✅ 需要宏录制功能  
· ✅ 系统需要高度可扩展

工控典型应用：

```cpp
// 工控操作序列
invoker.executeCommand(make_unique<MoveToHomeCommand>());
invoker.executeCommand(make_unique<CalibrateSensorCommand>());
invoker.executeCommand(make_unique<LoadRecipeCommand>("product_A"));

invoker.undoLastCommand();

// 批量执行检测命令
auto detectionBatch = make_unique<CompositeCommand>("QualityCheck");
detectionBatch->addCommand(make_unique<SetDetectionParamsCommand>(params));
detectionBatch->addCommand(make_unique<StartCaptureCommand>());
detectionBatch->addCommand(make_unique<AnalyzeImagesCommand>());

```

```cpp
// 结合两者优点
class HybridCaptureSystem {
private:
    std::unique_ptr<IImageCapture> capture;
    CaptureCommandInvoker commandInvoker;
    BasicController basicController;
    
public:
    // 简单操作使用控制器
    void quickSetup(CaptureMode mode) {
        basicController.switchMode(mode);
    }
    
    // 复杂操作使用命令模式
    void advancedConfiguration() {
        // 创建配置命令序列
        auto configCmd = std::make_unique<CompositeCommand>("AdvancedConfig");
        
        if (capture->getMode() == CaptureMode::ONLINE_DETECTION) {
            configCmd->addCommand(std::make_unique<SetModelCommand>("model.onnx"));
            configCmd->addCommand(std::make_unique<SetROICommand>(calculateROIs()));
            configCmd->addCommand(std::make_unique<SetThresholdCommand>(0.8));
        }
        
        commandInvoker.executeCommand(std::move(configCmd));
    }
    
    // 常用操作提供快捷方法（控制器封装）
    bool setModelForOnlineDetection(const std::string& path) {
        return basicController.setModelPath(path);
    }
    
    // 获取命令历史用于审计
    std::vector<std::string> getOperationHistory() const {
        return commandInvoker.getHistory();
    }
};
```

对于采图模块的推荐：

使用方案2（控制器封装），因为：

1. 模式只有2种（在线检测、预览）
2. 特殊操作数量有限
3. 不需要撤销功能
4. 工控系统更看重简洁和实时性

```cpp
class ImageCaptureController {
    // 提供简洁API
    bool setupOnlineDetection(const OnlineConfig& config) {
        switchToMode(ONLINE_DETECTION);
        return configureDetection(config); // 内部处理类型转换
    }

        switchToMode(PREVIEW);
        return configurePreview(config);

    bool configureDetection(const OnlineConfig& config) {
        auto detection = dynamic_cast<IOnlineDetection*>(currentCapture.get());
        if (!detection) return false;
        
        detection->setModelPath(config.modelPath);
```
