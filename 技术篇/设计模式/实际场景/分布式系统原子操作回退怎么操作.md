### <font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">分布式系统原子操作框架设计（基于两阶段提交协议）</font>
#### <font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">核心设计思路</font>
1. **两阶段提交协议（2PC）**<font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">：确保所有参与者要么全部提交，要么全部回滚</font>
2. **协调者-参与者模式**<font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">：中心协调者管理事务状态</font>
3. **操作可逆性**<font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">：所有子操作需提供逆向补偿操作</font>
4. **超时处理机制**<font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">：防止进程阻塞</font>

#### <font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">C++ 框架实现</font>
```cpp
#include <vector>
#include <functional>
#include <memory>
#include <atomic>
#include <stdexcept>

// 操作结果状态码
enum class OperationStatus {
    SUCCESS,
    FAILURE,
    TIMEOUT
};

// 原子操作抽象基类
class AtomicOperation {
public:
    virtual ~AtomicOperation() = default;
    virtual OperationStatus execute() = 0;
    virtual OperationStatus compensate() = 0;
    virtual std::string id() const = 0;
};

// 协调者类（核心控制器）
class TransactionCoordinator {
public:
    void add_operation(std::unique_ptr<AtomicOperation> op) {
        operations_.push_back(std::move(op));
    }

    bool execute_transaction() {
        // ========== 准备阶段 ==========
        std::vector<AtomicOperation*> prepared;
        
        for (auto& op : operations_) {
            switch (op->execute()) {
                case OperationStatus::SUCCESS:
                    prepared.push_back(op.get());
                    break;
                case OperationStatus::FAILURE:
                    rollback_operations(prepared);
                    return false;
                case OperationStatus::TIMEOUT:
                    handle_timeout(op.get());
                    rollback_operations(prepared);
                    return false;
            }
        }

        // ========== 提交阶段 ==========
        for (auto* op : prepared) {
            if (op->commit() != OperationStatus::SUCCESS) {
                // 提交失败时执行全局补偿
                global_compensation();
                return false;
            }
        }

        return true;
    }

private:
    void rollback_operations(const std::vector<AtomicOperation*>& ops) {
        for (auto it = ops.rbegin(); it != ops.rend(); ++it) {
            (*it)->compensate();
        }
    }

    void global_compensation() {
        for (auto it = operations_.rbegin(); it != operations_.rend(); ++it) {
            (*it)->compensate();
        }
    }

    void handle_timeout(AtomicOperation* op) {
        // 实际系统应包含重试/报警等逻辑
        op->compensate();
    }

    std::vector<std::unique_ptr<AtomicOperation>> operations_;
};
```

### <font style="color:rgb(38, 38, 38);background-color:rgb(239, 246, 255);">增强框架设计</font>
```cpp
#include <vector>
#include <functional>
#include <memory>
#include <atomic>
#include <stdexcept>
#include <map>
#include <fstream>
#include <chrono>
#include <thread>

// 事务状态（持久化记录）
enum class TransactionState {
    INIT,
    PREPARING,
    PREPARED,   // 所有操作准备成功
    COMMITTING,
    COMMITTED,
    ROLLING_BACK,
    ROLLED_BACK
};

// 操作状态
enum class OperationStatus {
    SUCCESS,
    FAILURE,
    TIMEOUT,
    UNKNOWN
};

// 日志管理类（持久化事务状态）
class TransactionLogger {
public:
    explicit TransactionLogger(const std::string& log_dir) 
        : log_dir_(log_dir) {
        // 确保日志目录存在
        std::filesystem::create_directories(log_dir);
    }
    
    void log_transaction_state(const std::string& tx_id, 
                              TransactionState state) {
        std::lock_guard<std::mutex> lock(mutex_);
        std::ofstream logfile(log_dir_ + "/" + tx_id + ".state", 
                             std::ios::app);
        logfile << static_cast<int>(state) << "\n";
    }
    
    void log_operation_prepared(const std::string& tx_id,
                               const std::string& op_id) {
        std::lock_guard<std::mutex> lock(mutex_);
        std::ofstream logfile(log_dir_ + "/" + tx_id + ".ops", 
                             std::ios::app);
        logfile << op_id << "\n";
    }
    
    TransactionState recover_state(const std::string& tx_id) {
        std::lock_guard<std::mutex> lock(mutex_);
        std::ifstream state_file(log_dir_ + "/" + tx_id + ".state");
        if (!state_file) return TransactionState::INIT;
        
        int last_state = 0;
        int state_val;
        while (state_file >> state_val) {
            last_state = state_val;
        }
        return static_cast<TransactionState>(last_state);
    }
    
    std::vector<std::string> recover_prepared_ops(const std::string& tx_id) {
        std::lock_guard<std::mutex> lock(mutex_);
        std::vector<std::string> ops;
        std::ifstream ops_file(log_dir_ + "/" + tx_id + ".ops");
        if (!ops_file) return ops;
        
        std::string op_id;
        while (std::getline(ops_file, op_id)) {
            if (!op_id.empty()) {
                ops.push_back(op_id);
            }
        }
        return ops;
    }
    
    void cleanup_logs(const std::string& tx_id) {
        std::lock_guard<std::mutex> lock(mutex_);
        std::remove((log_dir_ + "/" + tx_id + ".state").c_str());
        std::remove((log_dir_ + "/" + tx_id + ".ops").c_str());
    }

private:
    std::string log_dir_;
    std::mutex mutex_;
};

// 原子操作抽象基类（增强版）
class AtomicOperation {
public:
    virtual ~AtomicOperation() = default;
    
    // 准备操作（可逆）
    virtual OperationStatus prepare() = 0;
    
    // 确认提交（不可逆）
    virtual OperationStatus commit() = 0;
    
    // 回滚操作
    virtual OperationStatus rollback() = 0;
    
    // 查询操作状态
    virtual OperationStatus query_status() = 0;
    
    // 唯一标识符
    virtual std::string id() const = 0;
};

// 协调者类（支持崩溃恢复）
class TransactionCoordinator {
public:
    TransactionCoordinator(std::shared_ptr<TransactionLogger> logger)
        : logger_(logger), recovery_mode_(false) {
        // 生成唯一事务ID
        tx_id_ = generate_tx_id();
    }
    
    // 恢复模式构造函数
    TransactionCoordinator(std::shared_ptr<TransactionLogger> logger, 
                          const std::string& existing_tx_id)
        : logger_(logger), tx_id_(existing_tx_id), recovery_mode_(true) {
        // 恢复未完成事务
        recover_incomplete_transaction();
    }

    void add_operation(std::unique_ptr<AtomicOperation> op) {
        if (recovery_mode_) {
            throw std::runtime_error("Cannot add operations in recovery mode");
        }
        operations_.push_back(std::move(op));
    }

    bool execute_transaction() {
        if (recovery_mode_) {
            throw std::runtime_error("Transaction already in recovery mode");
        }
        
        // 记录初始状态
        logger_->log_transaction_state(tx_id_, TransactionState::INIT);
        
        try {
            // ====== 准备阶段 ======
            logger_->log_transaction_state(tx_id_, TransactionState::PREPARING);
            
            std::vector<AtomicOperation*> prepared;
            for (auto& op : operations_) {
                OperationStatus status = op->prepare();
                
                if (status == OperationStatus::SUCCESS) {
                    // 记录操作准备成功
                    logger_->log_operation_prepared(tx_id_, op->id());
                    prepared.push_back(op.get());
                } else {
                    // 准备阶段失败
                    logger_->log_transaction_state(tx_id_, TransactionState::ROLLING_BACK);
                    rollback_operations(prepared);
                    logger_->log_transaction_state(tx_id_, TransactionState::ROLLED_BACK);
                    return false;
                }
            }
            
            // 所有操作准备成功
            logger_->log_transaction_state(tx_id_, TransactionState::PREPARED);
            
            // ====== 提交阶段 ======
            logger_->log_transaction_state(tx_id_, TransactionState::COMMITTING);
            for (auto* op : prepared) {
                if (op->commit() != OperationStatus::SUCCESS) {
                    // 提交失败执行全局补偿
                    logger_->log_transaction_state(tx_id_, TransactionState::ROLLING_BACK);
                    global_rollback();
                    logger_->log_transaction_state(tx_id_, TransactionState::ROLLED_BACK);
                    return false;
                }
            }
            
            // 提交成功
            logger_->log_transaction_state(tx_id_, TransactionState::COMMITTED);
            cleanup_logs();
            return true;
        } catch (const std::exception& e) {
            // 发生异常时执行回滚
            logger_->log_transaction_state(tx_id_, TransactionState::ROLLING_BACK);
            global_rollback();
            logger_->log_transaction_state(tx_id_, TransactionState::ROLLED_BACK);
            return false;
        }
    }

private:
    void rollback_operations(const std::vector<AtomicOperation*>& ops) {
        for (auto it = ops.rbegin(); it != ops.rend(); ++it) {
            (*it)->rollback();
        }
    }

    void global_rollback() {
        // 获取所有已准备的操作
        auto prepared_ops = logger_->recover_prepared_ops(tx_id_);
        std::vector<AtomicOperation*> ops_to_rollback;
        
        // 查找对应的操作对象
        for (const auto& op_id : prepared_ops) {
            for (auto& op : operations_) {
                if (op->id() == op_id) {
                    ops_to_rollback.push_back(op.get());
                    break;
                }
            }
        }
        
        // 执行回滚
        rollback_operations(ops_to_rollback);
    }

    void recover_incomplete_transaction() {
        TransactionState state = logger_->recover_state(tx_id_);
        
        switch (state) {
            case TransactionState::PREPARED:
                // 准备完成后崩溃：提交事务
                commit_prepared();
                break;
                
            case TransactionState::COMMITTING:
                // 提交过程中崩溃：重试提交
                retry_commit();
                break;
                
            case TransactionState::ROLLING_BACK:
                // 回滚过程中崩溃：重试回滚
                retry_rollback();
                break;
                
            default:
                // 其他状态无需处理或已结束
                break;
        }
    }

    void commit_prepared() {
        logger_->log_transaction_state(tx_id_, TransactionState::COMMITTING);
        auto prepared_ops = logger_->recover_prepared_ops(tx_id_);
        bool all_committed = true;
        
        for (const auto& op_id : prepared_ops) {
            for (auto& op : operations_) {
                if (op->id() == op_id) {
                    // 查询操作当前状态
                    OperationStatus status = op->query_status();
                    
                    if (status == OperationStatus::UNKNOWN) {
                        // 状态未知，尝试提交
                        if (op->commit() != OperationStatus::SUCCESS) {
                            all_committed = false;
                        }
                    } else if (status != OperationStatus::SUCCESS) {
                        // 操作未成功提交
                        all_committed = false;
                    }
                    break;
                }
            }
        }
        
        if (all_committed) {
            logger_->log_transaction_state(tx_id_, TransactionState::COMMITTED);
            cleanup_logs();
        } else {
            logger_->log_transaction_state(tx_id_, TransactionState::ROLLING_BACK);
            global_rollback();
            logger_->log_transaction_state(tx_id_, TransactionState::ROLLED_BACK);
        }
    }

    void retry_commit() {
        // 与commit_prepared类似，但需处理部分已提交的情况
        commit_prepared();
    }

    void retry_rollback() {
        logger_->log_transaction_state(tx_id_, TransactionState::ROLLING_BACK);
        global_rollback();
        logger_->log_transaction_state(tx_id_, TransactionState::ROLLED_BACK);
        cleanup_logs();
    }

    void cleanup_logs() {
        logger_->cleanup_logs(tx_id_);
    }

    // 生成唯一事务ID
    std::string generate_tx_id() {
        static std::atomic<int> counter{0};
        auto now = std::chrono::system_clock::now();
        auto timestamp = std::chrono::duration_cast<std::chrono::milliseconds>(
            now.time_since_epoch()).count();
        return "TX_" + std::to_string(timestamp) + "_" + std::to_string(++counter);
    }

    std::vector<std::unique_ptr<AtomicOperation>> operations_;
    std::shared_ptr<TransactionLogger> logger_;
    std::string tx_id_;
    bool recovery_mode_;
};

class RecoveryManager {
public:
    RecoveryManager(const std::string& log_dir) 
        : logger_(std::make_shared<TransactionLogger>(log_dir)) {}
    
    void recover_all_transactions() {
        // 扫描日志目录中的所有事务状态文件
        for (const auto& entry : std::filesystem::directory_iterator(logger_->log_dir)) {
            if (entry.path().extension() == ".state") {
                std::string tx_id = entry.path().stem().string();
                TransactionState state = logger_->recover_state(tx_id);
                
                // 处理未完成的事务
                if (state == TransactionState::PREPARED || 
                    state == TransactionState::COMMITTING ||
                    state == TransactionState::ROLLING_BACK) {
                    
                    // 创建恢复协调者
                    TransactionCoordinator recovery_coord(logger_, tx_id);
                    // 恢复过程会自动执行
                }
            }
        }
    }

private:
    std::shared_ptr<TransactionLogger> logger_;
};

int main() {
    const std::string log_dir = "/var/log/transactions";
    
    // 系统启动时恢复未完成事务
    RecoveryManager recovery_manager(log_dir);
    recovery_manager.recover_all_transactions();
    
    // 创建新事务
    auto logger = std::make_shared<TransactionLogger>(log_dir);
    TransactionCoordinator coordinator(logger);
    
    // 添加操作（传递事务ID给远程操作）
```
