```cpp
// 抽象产品接口
class Shape {
public:
virtual void draw() = 0;
virtual ~Shape() {} // 虚析构确保正确释放资源
};

// 具体产品
class Circle : public Shape {
public:
void draw() override { cout << "绘制圆形" << endl; }
};

class Rectangle : public Shape {
public:
void draw() override { cout << "绘制矩形" << endl; }
};

// 简单工厂
class ShapeFactory {
public:
static Shape* createShape(const string& type) {
    if (type == "circle")    return new Circle();
    if (type == "rectangle") return new Rectangle();
    return nullptr;
}
};

// 客户端代码
int main() {
    string type;
    cout << "输入形状类型 (circle/rectangle): ";
    cin >> type;

    Shape* shape = ShapeFactory::createShape(type);
    if (shape) {
        shape->draw();
        delete shape;
    } else {
        cout << "无效类型!" << endl;
    }
    return 0;
}
```

### _<font style="color:rgb(64, 64, 64);">优化方案1：类型注册机制（编译时解耦）</font>_
#### _<font style="color:rgb(64, 64, 64);">实现代码</font>_
```cpp
#include <map>
#include <string>
#include <functional>
#include <memory>

class Car { /* 同前 */ };

// 产品注册表（关键基础设施）
class CarRegistry {
private:
    static std::map<std::string, std::function<Car*()>>& getMap() {
        static std::map<std::string, std::function<Car*()>> instances;
        return instances;
    }

public:
    static void registerCar(const std::string& name, auto&& func) {
        getMap()[name] = std::forward<decltype(func)>(func);
    }

    static Car* create(const std::string& name) {
        if(auto it = getMap().find(name); it != getMap().end())
            return it->second();
        throw std::runtime_error("未知车型");
    }
};

// 自动注册模板（每个车型只需写一次）
template<typename T>
struct AutoRegister {
    explicit AutoRegister(const std::string& name) {
        CarRegistry::registerCar(name, []{ return new T(); });
    }
};

// 具体车型定义（与客户端完全隔离）
namespace {
    class Benz : public Car {
    public:
        void drive() override { std::cout << "奔驰启动\n"; }
        static AutoRegister<Benz> registrar;  // 自动注册
    };
    AutoRegister<Benz> Benz::registrar("benz");

    class BMW : public Car {
    public:
        void drive() override { std::cout << "宝马启动\n"; }
        static AutoRegister<BMW> registrar;
    };
    AutoRegister<BMW> BMW::registrar("bmw");
}

// 客户端代码（零产品类依赖）
int main() {
    std::unique_ptr<Car> car(CarRegistry::create("benz"));
    car->drive();  // 输出：奔驰启动
}
```

#### _<font style="color:rgb(64, 64, 64);">关键创新点</font>_
1. **注册机制**_<font style="color:rgb(64, 64, 64);">：通过</font>_`_<font style="color:rgb(64, 64, 64);">CarRegistry</font>_`_<font style="color:rgb(64, 64, 64);">集中管理车型创建逻辑  </font>_
2. **自动注册**_<font style="color:rgb(64, 64, 64);">：利用模板在全局初始化阶段完成注册  </font>_
3. **匿名命名空间**_<font style="color:rgb(64, 64, 64);">：彻底隐藏具体产品类的存在</font>_

### _<font style="color:rgb(64, 64, 64);">优化方案2：静态接口+Concept（C++20 终极方案）</font>_
```cpp
#include <concepts>

// 定义Car概念（编译时检查）
template<typename T>
concept Car = requires(T t) {
    { t.drive() } -> std::same_as<void>;
};

// 通用工厂模板
template<Car T>
struct CarFactory {
    static T* create() { return new T(); }
};

// 客户端代码（仅依赖概念）
template<Car T>
void clientCode() {
    T* car = CarFactory<T>::create();
    car->drive();
    delete car;
}

// 具体车型定义（可放在独立模块）
class Benz {
public:
    void drive() { std::cout << "奔驰启动\n"; }
};

// 使用
int main() {
    clientCode<Benz>();  // ✅ 合法，Benz满足Car概念
    // clientCode<int>(); // ❌ 编译错误：不满足Car概念
}
```

#### _<font style="color:rgb(64, 64, 64);">优势分析</font>_
+ **零运行时开销**_<font style="color:rgb(64, 64, 64);">：全部在编译期解析  </font>_
+ **强类型约束**_<font style="color:rgb(64, 64, 64);">：通过Concept确保类型安全  </font>_
+ **极致解耦**_<font style="color:rgb(64, 64, 64);">：客户端只需知道概念，无需头文件依赖</font>_

### _<font style="color:rgb(64, 64, 64);">优化方案3：依赖注入容器（工业级解决方案）</font>_
```cpp
#include <boost/di.hpp>
namespace di = boost::di;

// 定义抽象接口
class ICar {
public:
    virtual void drive() = 0;
    virtual ~ICar() = default;
};

// 具体实现（与客户端隔离）
class Benz : public ICar { /*...*/ };
class BMW : public ICar { /*...*/ };

// 配置注入规则（通常在独立模块）
auto carModule = [] {
    return di::make_injector(
        di::bind<ICar>().to<Benz>()  // 配置默认车型
    );
};

// 客户端代码（仅依赖接口）
class CarUser {
public:
    explicit CarUser(std::shared_ptr<ICar> car) : car_(car) {}
    void use() { car_->drive(); }
private:
    std::shared_ptr<ICar> car_;
};

int main() {
    auto injector = carModule();
    auto user = injector.create<CarUser>();
    user.use();  // 输出：奔驰启动
}
```
