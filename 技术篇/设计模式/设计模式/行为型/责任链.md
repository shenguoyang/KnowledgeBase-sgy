责任链模式 (Chain of Responsibility Pattern) - C++ 实现

典型应用场景：请求审批系统

基础实现

```cpp
#include <iostream>
#include <string>
#include <memory>

// 请求类
class PurchaseRequest {
private:
    double amount_;
    std::string purpose_;
    
public:
    PurchaseRequest(double amount, const std::string& purpose) 
        : amount_(amount), purpose_(purpose) {}
    
    double getAmount() const { return amount_; }
    std::string getPurpose() const { return purpose_; }
};

// 抽象处理器
class Approver {
protected:
    std::string name_;
    std::shared_ptr<Approver> next_approver_;  // 下一个处理者
    
public:
    Approver(const std::string& name) : name_(name) {}
    virtual ~Approver() = default;
    
    // 设置下一个处理者
    void setNextApprover(std::shared_ptr<Approver> next) {
        next_approver_ = next;
    }
    
    // 处理请求的抽象方法
    virtual void processRequest(const PurchaseRequest& request) = 0;
    
protected:
    // 传递给下一个处理者
    void forwardRequest(const PurchaseRequest& request) {
        if (next_approver_) {
            next_approver_->processRequest(request);
        } else {
            std::cout << "请求无法处理，没有更高权限的审批人" << std::endl;
        }
    }
};

// 具体处理器 - 经理
class Manager : public Approver {
private:
    double approval_limit_ = 5000.0;
    
public:
    Manager(const std::string& name) : Approver(name) {}
    
    void processRequest(const PurchaseRequest& request) override {
        if (request.getAmount() <= approval_limit_) {
            std::cout << "经理 " << name_ << " 审批了采购请求: " 
                      << request.getPurpose() << ", 金额: " << request.getAmount() << std::endl;
        } else {
            std::cout << "经理 " << name_ << " 无权限审批，转交上级" << std::endl;
            forwardRequest(request);
        }
    }
};

// 具体处理器 - 总监
class Director : public Approver {
private:
    double approval_limit_ = 10000.0;
    
public:
    Director(const std::string& name) : Approver(name) {}
    
    void processRequest(const PurchaseRequest& request) override {
        if (request.getAmount() <= approval_limit_) {
            std::cout << "总监 " << name_ << " 审批了采购请求: " 
                      << request.getPurpose() << ", 金额: " << request.getAmount() << std::endl;
        } else {
            std::cout << "总监 " << name_ << " 无权限审批，转交上级" << std::endl;
            forwardRequest(request);
        }
    }
};

// 具体处理器 - 副总裁
class VicePresident : public Approver {
private:
    double approval_limit_ = 50000.0;
    
public:
    VicePresident(const std::string& name) : Approver(name) {}
    
    void processRequest(const PurchaseRequest& request) override {
        if (request.getAmount() <= approval_limit_) {
            std::cout << "副总裁 " << name_ << " 审批了采购请求: " 
                      << request.getPurpose() << ", 金额: " << request.getAmount() << std::endl;
        } else {
            std::cout << "副总裁 " << name_ << " 无权限审批，转交上级" << std::endl;
            forwardRequest(request);
        }
    }
};

// 具体处理器 - 总裁
class President : public Approver {
public:
    President(const std::string& name) : Approver(name) {}
    
    void processRequest(const PurchaseRequest& request) override {
        std::cout << "总裁 " << name_ << " 审批了采购请求: " 
                  << request.getPurpose() << ", 金额: " << request.getAmount() << std::endl;
    }
};

// 使用示例
int main() {
    // 创建责任链
    auto manager = std::make_shared<Manager>("张经理");
    auto director = std::make_shared<Director>("李总监");
    auto vp = std::make_shared<VicePresident>("王副总裁");
    auto president = std::make_shared<President>("陈总裁");
    
    // 设置责任链
    manager->setNextApprover(director);
    director->setNextApprover(vp);
    vp->setNextApprover(president);
    
    // 创建请求
    PurchaseRequest request1(3000, "购买办公用品");
    PurchaseRequest request2(8000, "购买服务器");
    PurchaseRequest request3(30000, "公司车辆采购");
    PurchaseRequest request4(100000, "新办公室装修");
    
    // 处理请求
    std::cout << "=== 采购审批流程 ===" << std::endl;
    manager->processRequest(request1);
    std::cout << "---" << std::endl;
    manager->processRequest(request2);
    std::cout << "---" << std::endl;
    manager->processRequest(request3);
    std::cout << "---" << std::endl;
    manager->processRequest(request4);
    
    return 0;
}
```

高级实现：支持动态配置和条件判断

```cpp
#include <iostream>
#include <string>
#include <memory>
#include <vector>
#include <functional>

// 更复杂的请求类
class Request {
public:
    enum class Type { PURCHASE, LEAVE, EXPENSE, HIRING };
    
private:
    Type type_;
    double amount_;
    std::string description_;
    std::string department_;
    int urgency_level_;  // 紧急程度 1-5
    
public:
    Request(Type type, double amount, const std::string& desc, 
            const std::string& dept = "", int urgency = 1)
        : type_(type), amount_(amount), description_(desc), 
          department_(dept), urgency_level_(urgency) {}
    
    Type getType() const { return type_; }
    double getAmount() const { return amount_; }
    std::string getDescription() const { return description_; }
    std::string getDepartment() const { return department_; }
    int getUrgencyLevel() const { return urgency_level_; }
};

// 高级抽象处理器
class Handler {
protected:
    std::string name_;
    std::shared_ptr<Handler> next_handler_;
    
public:
    Handler(const std::string& name) : name_(name) {}
    virtual ~Handler() = default;
    
    void setNextHandler(std::shared_ptr<Handler> next) {
        next_handler_ = next;
    }
    
    // 支持条件判断的处理方法
    virtual bool canHandle(const Request& request) = 0;
    virtual void handleRequest(const Request& request) = 0;
    
    // 模板方法 - 定义处理流程
    void process(const Request& request) {
        if (canHandle(request)) {
            handleRequest(request);
        } else if (next_handler_) {
            std::cout << name_ << " 无法处理，转交给下一处理者" << std::endl;
            next_handler_->process(request);
        } else {
            std::cout << "请求无法处理: " << request.getDescription() << std::endl;
        }
    }
};

// 具体处理器 - 部门经理
class DepartmentManager : public Handler {
public:
    DepartmentManager(const std::string& name) : Handler(name) {}
    
    bool canHandle(const Request& request) override {
        // 部门经理可以处理本部门的常规请求
        return (request.getDepartment() == "技术部" && 
                request.getAmount() <= 5000 &&
                request.getUrgencyLevel() <= 3);
    }
    
    void handleRequest(const Request& request) override {
        std::cout << "部门经理 " << name_ << " 处理了请求: " 
                  << request.getDescription() << std::endl;
    }
};

// 具体处理器 - 财务经理
class FinanceManager : public Handler {
public:
    FinanceManager(const std::string& name) : Handler(name) {}
    
    bool canHandle(const Request& request) override {
        // 财务经理处理所有财务相关请求
        return (request.getType() == Request::Type::PURCHASE || 
                request.getType() == Request::Type::EXPENSE) &&
               request.getAmount() <= 20000;
    }
    
    void handleRequest(const Request& request) override {
        std::cout << "财务经理 " << name_ << " 审批了财务请求: " 
                  << request.getDescription() << ", 金额: " << request.getAmount() << std::endl;
    }
};

// 具体处理器 - HR经理
class HRManager : public Handler {
public:
    HRManager(const std::string& name) : Handler(name) {}
    
    bool canHandle(const Request& request) override {
        // HR经理处理人事相关请求
        return (request.getType() == Request::Type::LEAVE || 
                request.getType() == Request::Type::HIRING);
    }
    
    void handleRequest(const Request& request) override {
        std::cout << "HR经理 " << name_ << " 处理了人事请求: " 
                  << request.getDescription() << std::endl;
    }
};

// 具体处理器 - 总经理
class GeneralManager : public Handler {
public:
    GeneralManager(const std::string& name) : Handler(name) {}
    
    bool canHandle(const Request& request) override {
        // 总经理处理所有重要请求
        return request.getAmount() <= 100000 || 
               request.getUrgencyLevel() >= 4;
    }
    
    void handleRequest(const Request& request) override {
        std::cout << "总经理 " << name_ << " 处理了重要请求: " 
                  << request.getDescription() << std::endl;
    }
};

// 责任链构建器
class HandlerChainBuilder {
private:
    std::shared_ptr<Handler> first_;
    std::shared_ptr<Handler> current_;
    
public:
    HandlerChainBuilder& addHandler(std::shared_ptr<Handler> handler) {
        if (!first_) {
            first_ = handler;
            current_ = handler;
        } else {
            current_->setNextHandler(handler);
            current_ = handler;
        }
        return *this;
    }
    
    std::shared_ptr<Handler> build() {
        return first_;
    }
};

// 使用示例
int main() {
    std::cout << "=== 高级责任链模式示例 ===" << std::endl;
    
    // 使用构建器创建责任链
    auto chain = HandlerChainBuilder()
        .addHandler(std::make_shared<DepartmentManager>("张经理"))
        .addHandler(std::make_shared<FinanceManager>("李财务"))
        .addHandler(std::make_shared<HRManager>("王HR"))
        .addHandler(std::make_shared<GeneralManager>("陈总"))
        .build();
    
    // 创建各种请求
    Request techRequest(Request::Type::PURCHASE, 3000, "购买开发设备", "技术部", 2);
    Request financeRequest(Request::Type::EXPENSE, 15000, "季度市场费用", "市场部", 1);
    Request hrRequest(Request::Type::HIRING, 0, "招聘高级工程师", "技术部", 3);
    Request urgentRequest(Request::Type::PURCHASE, 50000, "紧急服务器采购", "技术部", 5);
    Request largeRequest(Request::Type::PURCHASE, 200000, "新办公室租赁", "行政部", 1);
    
    // 处理请求
    std::cout << "\n1. 技术部采购请求:" << std::endl;
    chain->process(techRequest);
    
    std::cout << "\n2. 市场部费用请求:" << std::endl;
    chain->process(financeRequest);
    
    std::cout << "\n3. 招聘请求:" << std::endl;
    chain->process(hrRequest);
    
    std::cout << "\n4. 紧急采购请求:" << std::endl;
    chain->process(urgentRequest);
    
    std::cout << "\n5. 大额采购请求:" << std::endl;
    chain->process(largeRequest);
    
    return 0;
}
```

责任链模式的优缺点

优点

1. 降低耦合度

```cpp
// 发送者不需要知道具体的处理者
handler->process(request); // 不需要知道是哪个具体handler处理
```

2. 增强灵活性

```cpp
// 可以动态改变责任链
chainBuilder.addHandler(newHandler).removeHandler(oldHandler);
```

3. 简化对象职责

```cpp
// 每个处理者只关心自己能否处理，不关心链的结构
bool canHandle(const Request& request) {
    return request.getAmount() <= my_limit;
}
```

4. 符合开闭原则

```cpp
// 新增处理者不需要修改现有代码
class NewHandler : public Handler {
    // 实现canHandle和handleRequest
};
// 只需添加到链中即可
```

缺点

1. 请求可能未被处理

```cpp
// 如果链中没有合适的处理者，请求可能丢失
if (!next_handler_) {
    // 请求未被处理
}
```

2. 性能问题

```cpp
// 长链可能导致性能下降
// 链: A -> B -> C -> D -> E
// 请求需要遍历整个链
```

3. 调试困难

```cpp
// 难以跟踪请求的处理路径
// 请求在多个对象间传递，调试复杂
```

4. 可能产生循环引用

```cpp
// 如果设置不当，可能导致循环链
handlerA->setNext(handlerB);
handlerB->setNext(handlerA); // 循环！
```

适用场景

1. 多个对象可以处理同一请求  
· 如：审批系统、异常处理、事件处理
2. 需要动态指定处理者  
· 如：根据运行时条件决定处理流程
3. 不希望明确指定接收者  
· 如：GUI事件处理、中间件
4. 处理流程可能变化  
· 如：工作流系统、过滤器链

责任链模式通过将请求的发送者和接收者解耦，提供了很好的灵活性，但需要注意链的长度和请求处理的可靠性问题。
