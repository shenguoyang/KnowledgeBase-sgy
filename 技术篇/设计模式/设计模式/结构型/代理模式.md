代理模式是一种结构型设计模式，它为其他对象提供一种代理以控制对这个对象的访问。

核心概念

代理模式通过引入一个代理对象来间接访问目标对象，从而可以在不改变目标对象的情况下，增加额外的功能或控制访问。

三种主要类型

1. 虚拟代理（延迟加载）

```cpp
// 接口
class Image {
public:
    virtual ~Image() = default;
    virtual void display() = 0;
};

// 真实对象 - 加载开销大
class HighResolutionImage : public Image {
private:
    std::string filename;
    
    void loadFromDisk() {
        std::cout << "Loading heavy image: " << filename << std::endl;
        // 模拟耗时操作
        std::this_thread::sleep_for(std::chrono::seconds(2));
    }
    
public:
    HighResolutionImage(const std::string& file) : filename(file) {
        loadFromDisk();
    }
    
    void display() override {
        std::cout << "Displaying image: " << filename << std::endl;
    }
};

// 虚拟代理 - 延迟加载
class ImageProxy : public Image {
private:
    std::string filename;
    mutable std::unique_ptr<HighResolutionImage> realImage; // 延迟创建
    
    void loadRealImage() const {
        if (!realImage) {
            realImage = std::make_unique<HighResolutionImage>(filename);
        }
    }
    
public:
    ImageProxy(const std::string& file) : filename(file) {}
    
    void display() override {
        loadRealImage(); // 真正需要时才加载
        realImage->display();
    }
};

// 使用
void demoVirtualProxy() {
    // 创建代理很快，真实对象尚未加载
    auto image = std::make_unique<ImageProxy>("large_photo.jpg");
    
    // 只有当真正显示时才会加载真实对象
    std::cout << "Proxy created, real image not loaded yet..." << std::endl;
    image->display(); // 此时才会加载真实图片
}
```

2. 保护代理（访问控制）

```cpp
// 敏感操作接口
class Database {
public:
    virtual ~Database() = default;
    virtual void query(const std::string& sql) = 0;
    virtual void update(const std::string& sql) = 0;
};

// 真实数据库
class RealDatabase : public Database {
public:
    void query(const std::string& sql) override {
        std::cout << "Executing query: " << sql << std::endl;
    }
    
    void update(const std::string& sql) override {
        std::cout << "Executing update: " << sql << std::endl;
    }
};

// 用户权限
enum class UserRole {
    GUEST,
    USER, 
    ADMIN
};

// 保护代理 - 权限控制
class ProtectedDatabase : public Database {
private:
    RealDatabase realDB;
    UserRole currentUser;
    
public:
    ProtectedDatabase(UserRole role) : currentUser(role) {}
    
    void query(const std::string& sql) override {
        // 所有角色都可以查询
        realDB.query(sql);
    }
    
    void update(const std::string& sql) override {
        // 只有管理员可以更新
        if (currentUser == UserRole::ADMIN) {
            realDB.update(sql);
        } else {
            std::cout << "Access denied: Update operation requires admin privileges" << std::endl;
        }
    }
};

// 使用
void demoProtectionProxy() {
    // 普通用户
    auto userDB = std::make_unique<ProtectedDatabase>(UserRole::USER);
    userDB->query("SELECT * FROM users");  // 允许
    userDB->update("DELETE FROM users");   // 拒绝
    
    // 管理员
    auto adminDB = std::make_unique<ProtectedDatabase>(UserRole::ADMIN);
    adminDB->update("DELETE FROM users");  // 允许
}
```

3. 远程代理（网络通信）

```cpp
// 通用接口
class FileSystem {
public:
    virtual ~FileSystem() = default;
    virtual std::string readFile(const std::string& path) = 0;
    virtual void writeFile(const std::string& path, const std::string& content) = 0;
};

// 远程服务（假设在另一台机器上）
class RemoteFileSystem : public FileSystem {
public:
    std::string readFile(const std::string& path) override {
        // 模拟网络调用
        std::cout << "Network call: reading " << path << " from remote server" << std::endl;
        return "File content from remote";
    }
    
    void writeFile(const std::string& path, const std::string& content) override {
        std::cout << "Network call: writing to " << path << " on remote server" << std::endl;
    }
};

// 远程代理 - 隐藏网络细节
class RemoteProxy : public FileSystem {
private:
    std::unique_ptr<RemoteFileSystem> remoteFS;
    std::unordered_map<std::string, std::string> cache; // 本地缓存
    
    // 模拟网络连接检查
    bool checkConnection() {
        std::cout << "Checking network connection..." << std::endl;
        return true; // 简化实现
    }
    
public:
    RemoteProxy() : remoteFS(std::make_unique<RemoteFileSystem>()) {}
    
    std::string readFile(const std::string& path) override {
        // 检查缓存
        if (cache.find(path) != cache.end()) {
            std::cout << "Returning cached file: " << path << std::endl;
            return cache[path];
        }
        
        // 网络操作
        if (checkConnection()) {
            auto content = remoteFS->readFile(path);
            cache[path] = content; // 缓存结果
            return content;
        }
        throw std::runtime_error("Network unavailable");
    }
    
    void writeFile(const std::string& path, const std::string& content) override {
        if (checkConnection()) {
            remoteFS->writeFile(path, content);
            cache[path] = content; // 更新缓存
        } else {
            throw std::runtime_error("Network unavailable");
        }
    }
};
```

智能指针中的代理模式

C++ 标准库中的 std::shared_ptr 和 std::weak_ptr 也使用了代理模式：

```cpp
void demoSmartPointerProxy() {
    class ExpensiveResource {
    public:
        ExpensiveResource() { std::cout << "Resource created\n"; }
        ~ExpensiveResource() { std::cout << "Resource destroyed\n"; }
        void use() { std::cout << "Using resource\n"; }
    };
    
    // shared_ptr 代理真实资源，控制生命周期
    auto resource = std::make_shared<ExpensiveResource>();
    
    // weak_ptr 作为观察者代理，不增加引用计数
    std::weak_ptr<ExpensiveResource> weakResource = resource;
    
    if (auto locked = weakResource.lock()) {
        locked->use();  // 通过代理访问
    }
}
```

实际应用场景

1. 日志代理

```cpp
class LoggingProxy : public Database {
private:
    std::unique_ptr<Database> realDB;
    
public:
    LoggingProxy(std::unique_ptr<Database> db) : realDB(std::move(db)) {}
    
    void query(const std::string& sql) override {
        std::cout << "[LOG] Query: " << sql << std::endl;
        auto start = std::chrono::high_resolution_clock::now();
        
        realDB->query(sql);
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
        std::cout << "[LOG] Query completed in " << duration.count() << "ms" << std::endl;
    }
};
```

代理模式 vs 装饰器模式

特性 代理模式 装饰器模式  
目的 控制访问 增强功能  
关系 代理知道真实对象的存在 装饰器与组件遵循相同接口  
关注点 访问机制、生命周期 动态添加职责

代理模式的核心价值：在不修改原始对象的情况下，通过代理层来控制访问、增加横切关注点（日志、缓存、权限等），实现关注点分离。
